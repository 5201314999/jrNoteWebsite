 function render(page){
     return new Promise((resolve,reject)=>{
         let viewUrl=`./views/${page}`;
         fs.readFile(viewUrl,'binary',(err,data)=>{
             if(err){
                 reject(err)
             }
             else{
                 resolve(data)
             }
         })
     })
 }
